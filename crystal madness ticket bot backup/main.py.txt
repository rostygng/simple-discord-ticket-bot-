import discord
from discord.ext import commands
from discord import app_commands
from datetime import datetime, timezone
import json
import asyncio

# ---------- Load bot token ----------
with open("bot_config/token.json") as f:
    secret_file = json.load(f)
TOKEN = secret_file["token"]

# ---------- Bot setup ----------
intents = discord.Intents.all()
bot = commands.Bot(command_prefix="!", intents=intents)

# ---------- Ticket Dropdown ----------
class TicketDropdown(discord.ui.Select):
    def __init__(self):
        options = [
            discord.SelectOption(label="Test Application", description="Application to be tested"),
            discord.SelectOption(label="Player report", description="Report a player that has broken the rules"),
            discord.SelectOption(label="Bug report", description="If there is a glitch or a server bug"),
            discord.SelectOption(label="Support", description="Discord or MC Server support"),
            discord.SelectOption(label="Questions", description="General Questions"),
            discord.SelectOption(label="Partners", description="If you want to partner with our server"),
            discord.SelectOption(label="Store", description="Shop support"),
        ]
        super().__init__(placeholder="Select a category...", options=options)

    async def callback(self, interaction: discord.Interaction):
        selected = self.values[0]

        # ---------- ACKNOWLEDGE IMMEDIATELY ----------
        await interaction.response.defer(ephemeral=True)

        guild = interaction.guild

        # ---------- Ensure Tickets category exists ----------
        category_name = "Tickets"
        category = discord.utils.get(guild.categories, name=category_name)
        if category is None:
            category = await guild.create_category(
                category_name,
                overwrites={guild.default_role: discord.PermissionOverwrite(read_messages=False)}
            )

        # ---------- Overwrites for ticket channel ----------
        overwrites = {
            guild.default_role: discord.PermissionOverwrite(read_messages=False),
            interaction.user: discord.PermissionOverwrite(read_messages=True, send_messages=True)
        }

        # ---------- Determine next ticket number ----------
        existing = [c.name for c in category.channels if c.name.startswith("ticket-")]
        ticket_numbers = [int(c.split("-")[1]) for c in existing if c.split("-")[1].isdigit()]
        next_number = max(ticket_numbers, default=0) + 1
        ticket_name = f"ticket-{next_number:03d}"

        ticket_channel = await guild.create_text_channel(
            name=ticket_name,
            overwrites=overwrites,
            category=category,
            topic=str(interaction.user.id)
        )

        # ---------- Embed ----------
        embed = discord.Embed(
            title=f"🎫 Ticket: {selected}",
            description=f"Hi {interaction.user.mention}, your ticket has been opened!\n\n**Category selected:** {selected}",
            color=discord.Color.green(),
            timestamp=datetime.now(timezone.utc)
        )

        # ---------- Ticket Buttons ----------
        class TicketButtons(discord.ui.View):
            def __init__(self, opener_id, channel):
                super().__init__(timeout=None)
                self.opener_id = opener_id
                self.channel = channel

            @discord.ui.button(label="Close Ticket", style=discord.ButtonStyle.red)
            async def close_ticket(self, button: discord.ui.Button, interaction_button: discord.Interaction):
                if interaction_button.user.id != self.opener_id:
                    await interaction_button.response.send_message("You cannot close this ticket.", ephemeral=True)
                    return
                await interaction_button.response.send_message(
                    "Ticket closed. You cannot send messages anymore. Click 'Delete Ticket' to remove it.",
                    ephemeral=False
                )
                await self.channel.set_permissions(interaction_button.guild.default_role, send_messages=False)
                await self.channel.set_permissions(interaction_button.user, send_messages=False)

            @discord.ui.button(label="Delete Ticket", style=discord.ButtonStyle.grey)
            async def delete_ticket(self, button: discord.ui.Button, interaction_button: discord.Interaction):
                if interaction_button.user.id != self.opener_id:
                    await interaction_button.response.send_message("You cannot delete this ticket.", ephemeral=True)
                    return
                await interaction_button.response.defer()
                await asyncio.sleep(1)
                await self.channel.delete()

        await ticket_channel.send(embed=embed, view=TicketButtons(interaction.user.id, ticket_channel))
        await interaction.followup.send(f"Your ticket has been created: {ticket_channel.mention}", ephemeral=True)

# ---------- Ticket View ----------
class TicketView(discord.ui.View):
    def __init__(self):
        super().__init__(timeout=None)
        self.add_item(TicketDropdown())

# ---------- /ticketpanel ----------
@bot.tree.command(name="ticketpanel", description="Create a ticket panel for support")
@app_commands.default_permissions(manage_guild=True)
async def ticketpanel(interaction: discord.Interaction):
    banner_url = "https://i.imgur.com/rostygnng.png"  # Replace with your banner URL

    embed = discord.Embed(
        title="🎫 Support Ticket Panel",
        description="Welcome! Please select a category from the dropdown below to open a ticket.",
        color=discord.Color.blue(),
        timestamp=datetime.now(timezone.utc)
    )

    if interaction.guild.icon:
        embed.set_thumbnail(url=interaction.guild.icon.url)

    embed.set_image(url=banner_url)
    await interaction.response.send_message(embed=embed, view=TicketView())

# ---------- Bot ready ----------
@bot.event
async def on_ready():
    print(f"Logged in as {bot.user} ({bot.user.id})")
    try:
        synced = await bot.tree.sync()
        print(f"Synced {len(synced)} command(s)")
    except Exception as e:
        print(e)

# ---------- Run bot ----------
bot.run(TOKEN)
